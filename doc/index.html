                var kjs = new Keybinds();
                
                kjs.init(function(){

                    kjs.bind("w","none", function(){
                       // Moves up
                    });

                    kjs.bind("a","none", function(){
                        //moves left
                    });

                    kjs.bind("s","none", function(){
                        //moves down
                    });

                    kjs.bind("d","none", function(){
                        //moves right;
                    });

                });
